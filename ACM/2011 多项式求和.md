### 杭电 2011 多项式求和



#### code:

```c
#include<stdio.h>
double fun(int n) {
	double sum = 0;
	for (int i = 1; i <= n; i++) {
		sum += i % 2 == 0 ? -1.0 / i : 1.0 / i;
	}
	return sum;
}
int main() {
	int n,m;
	scanf("%d", &m);
	for (int i = 0; i < m; i++) {
		scanf("%d", &n);
		printf("%.2f\n", fun(n));
	}
}
```
