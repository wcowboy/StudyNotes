### 杭电 2012 素数判定



code：

```c
#include<stdio.h>
#include<math.h>
bool fun(int n) {
	int i;
	for (i = 2; i < (int)sqrt((double)n); i++) 
		if (n % i == 0)
			break;
	return i == (int)sqrt((double)n) ? true : false;
}
int main() {
	int x, y, i;
	while (scanf("%d%d", &x, &y) != EOF && (x != 0 || y != 0)) {
		for (i = x; i <= y; i++) {
			if (!fun(i * i + i + 41))
				break;
		}
		i == (y + 1) ? printf("OK\n"): printf("Sorry\n");;
	}
}
```
