### 杭电2010 水仙花数



#### code:

```c
#include<stdio.h>
bool fun(int n) {
	int a, b, c;
	a = n / 100;
	b = n / 10 % 10;
	c = n % 10;
	return n == (a * a * a + b * b * b + c * c * c) ? true : false;
}
int main() {
	int n, m,sign;
	while (scanf("%d%d", &n, &m) != EOF) {
		sign = 0;
		for (int i = n; i <= m; i++) {
			if (fun(i) && sign == 0) {
				sign = 1;
				printf("%d", i);
				continue;
			}
			if(fun(i) && sign == 1)
				printf(" %d", i);
		}
		sign == 0 ? printf("no\n") : printf("\n");
	}
}
```
