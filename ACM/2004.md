### 2004 成绩转换



#### code:

```c
#include<stdio.h>
int main() {
	int n;
	while (scanf("%d", &n) != EOF) {
		if (n > 100 || n < 0) {
			printf("Score is error!\n");
			continue;
		}
		switch (n / 10){
		case 10:
		case 9: {
			printf("A\n");
			break;
		}
		case 8: {
			printf("B\n");
			break;
		}
		case 7: {
			printf("C\n");
			break;
		}
		case 6: {
			printf("D\n");
			break;
		}
		default:
			printf("E\n");
			break;
		}
	}
}
```
